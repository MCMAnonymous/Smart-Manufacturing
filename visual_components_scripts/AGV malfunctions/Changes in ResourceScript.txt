  def exportFailuresInfo(list_failures):
    global fileCheck, comp
    '''
    This method writes list_failures in a csv file in the output folder 
    '''
        
    # name of the folder in which CSVs are stored
    output_dir = "CSV"
    current_dir = os.getcwd()
    path = os.path.join(current_dir, output_dir)
    
    # data_Failures_CompName.csv columns name 
    columns_failures=['TimeStamp','Component', 'Failure','Broken']
    
    if not os.path.exists(path):
      os.makedirs(path)
      
    if fileCheck == 0: # first access to the file
      f = open(os.path.join(path, "data_Failures_%s.csv"%(comp.Name)), "w")
    
      # delimiter used in the CSV file
      failure_writer = csv.writer(f, delimiter=",", lineterminator='\n')
    
      failure_writer.writerow(columns_failures)
      
      for fila in list_failures:
        failure_writer.writerow(fila)
      
      list_failures[:] = []
      fileCheck += 1
      f.close()
    else:   
      f = open(os.path.join(path, "data_Failures_%s.csv"%(comp.Name)), "a")
      
      # delimiter used in the CSV file
      failure_writer = csv.writer(f, delimiter=",", lineterminator='\n')
      
      for fila in list_failures:
        failure_writer.writerow(fila)
      
      list_failures[:] = []
      f.close()
